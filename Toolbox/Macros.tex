% custom abstract environnment
% =============================================================
\newcommand\abstractname{Abstract} 
\makeatletter
  \newenvironment{abstract}{%
        \medskip
        \begin{center}%
          {\bfseries \abstractname\vspace{-.5em}\vspace{\z@}}%
        \end{center}%
        \quotation
      }
\makeatother

\def\@custommaketitle{%
  % \newpage
  \null
  \vskip 1em%
  \begin{center}%
  \let \footnote \thanks
    {\LARGE \@title \par}%
    \vskip 1.5em%
    {\large
      \lineskip .5em%
      % \begin{tabular}[t]{c}%
        \@author
      % \end{tabular}
	  \par}%
    \vskip 1em%
  \par
    {\large
      \lineskip .5em%
        \noindent \@affiliation
	  \par}%
    \vskip 1em%
  \par
    {\large
      \lineskip .5em%
        \noindent \@pubStatus
	  \par}%
    \vskip 1em%
  \end{center}%
  \par
  \begin{abstract}
     \@myabstract
  \end{abstract}
  \vskip 1.5em
	\par
	\keywordDisplay{\@keywords}
	\par
	\mathsubclassDisplay{\@mathsubclass}
}
