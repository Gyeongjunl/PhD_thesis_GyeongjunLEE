% chapter quote

% chapter abstract
\newenvironment{chaptabstract}[1]{%
\begin{center}
	\begin{minipage}{\textwidth}
		\noindent #1

\end{center}
}

	\end{minipage}
	\vspace*{\fill}
\end{center}
}
